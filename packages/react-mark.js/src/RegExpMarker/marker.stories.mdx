# ⚡️ `<RegExpMarker />`
A wrapper component to higlight strings matching a regular expression.

export const Template = (args) => <RegExpMarker {...args} />
const args = {
    children: [
        "Cillum proident eu eiusmod incididunt pariatur. ",
        "Ullamco qui deserunt ut reprehenderit cupidatat ",
        "cupidatat nisi non occaecat non commodo. ",
        "Magna incididunt eu laboris laboris labore. "
    ],
    mark: /qui/
}

## ‣ Configure Wrapper Element
This will render the wrapper element as a section instead
of the default `<div />`.
<Canvas>
    <Story 
        name="Section Element Wrapper" 
        args={{
            ...args, 
            as: "section"
        }}
    >
        {args => <RegExpMarker {...args} />}
    </Story>
</Canvas>
